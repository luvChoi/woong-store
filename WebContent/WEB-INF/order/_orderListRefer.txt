$(function(){ //이 페이지가 다 로딩되면 아래내용을 실행
	$("#btnLogin").click(function() {
		id = $("#id").val();
		passwd = $("#passwd").val();
		param = {"id":id, "passwd":passwd};
		$.ajax({ //비동기식
			type: "post",
			data: param,
			url: "test08Proc.do",
			success: function(data) {
				$("#result").html(data);
			}
		});
	});
});
